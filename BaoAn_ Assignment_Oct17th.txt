18. Tìm số hóa đơn trong năm 2006 đã mua ít nhất tất cả các sản phẩm do Singapore sản xuất.
SELECT SOHD, COUNT(MASP)
FROM SANPHAM JOIN CTHD ON SANPHAM.MASP = CTHD.MASP
WHERE NUOCSX = 'Singapore'
GROUP BY SOHD
HAVING COUNT(MASP) <= ALL (
	SELECT COUNT(MASP)
	FROM SANPHAM, CTHD
	WHERE SANPHAM JOIN CTHD ON SANPHAM.MASP = CTHD.MASP
	AND NUOCSX = 'Singapore'
	GROUP BY SOHD
)

41. 
SELECT TOP 1 WITH TIES CTHD.MASP, TENSP, SUM(SL)
FROM SANPHAM, CTHD, HOADON
WHERE CTHD.SOHD = HOADON.SOHD AND CTHD.MASP = SANPHAM.MASP
AND YEAR(NGDH) = 2006
GROUP BY CTHD.MASP, TENSP
HAVING SUM(SL) <= ALL(
	SELECT SUM(SL)
	FROM SANPHAM, CTHD, HOADON
	WHERE CTHD.SOHD = HOADON.SOHD AND CTHD.MASP = SANPHAM.MASP AND YEAR(NGDH) = 2006
	GROUP BY CTHD.MASP, TENSP
)
